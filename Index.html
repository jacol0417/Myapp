    <div id="app" class="max-w-2xl mx-auto bg-white rounded-xl shadow-md overflow-hidden p-6">

       

        <div class="flex flex-wrap justify-between items-center gap-4 mb-6 border-b pb-4">

            <h1 class="text-2xl font-bold text-blue-600">📋 SOP 備忘錄 (跨平台版)</h1>

            <div class="flex gap-2">

                <label class="bg-green-600 hover:bg-green-700 text-white px-3 py-1.5 rounded-lg text-sm cursor-pointer shadow">

                    📥 匯入 CSV

                    <input type="file" accept=".csv" class="hidden" @change="handleImportCSV">

                </label>

                <button @click="handleExportCSV" class="bg-blue-600 hover:bg-blue-700 text-white px-3 py-1.5 rounded-lg text-sm cursor-pointer shadow">

                    📤 匯出 CSV

                </button>

            </div>

        </div>

        <div class="flex gap-3 mb-4">

            <input type="text" v-model="keyword" placeholder="🔍 輸入關鍵字搜尋..."

                   class="flex-1 px-4 py-2 border rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500 text-sm">

            <button @click="isFormVisible = !isFormVisible" class="bg-gray-500 hover:bg-gray-600 text-white px-4 py-2 rounded-lg text-sm transition shadow">

                {{ isFormVisible ? '🙈 隱藏表單' : '✏️ 開啟表單' }}

            </button>

        </div>

        <div v-show="isFormVisible" class="bg-gray-50 p-4 rounded-xl mb-6 border transition-all">

            <h2 class="text-lg font-semibold mb-3 text-gray-700">

                {{ editingId ? '🔄 編輯項目' : '💾 新增 SOP 項目' }}

            </h2>

            <div class="grid grid-cols-2 gap-3">

                <input type="text" v-model="form.name" placeholder="大分類 (如: 電工維修) *" class="p-2 border rounded text-sm">

                <input type="text" v-model="form.subName" placeholder="小分類 (如: 常用工具) *" class="p-2 border rounded text-sm">

                <input type="text" v-model="form.content" placeholder="詳細內容 *" class="p-2 border rounded text-sm col-span-2">

                <input type="text" v-model="form.remark" placeholder="備註" class="p-2 border rounded text-sm col-span-2">

            </div>

            <div class="flex justify-end gap-2 mt-4">

                <button @click="clearForm" class="bg-gray-300 hover:bg-gray-400 text-gray-700 px-4 py-1.5 rounded text-sm">清空</button>

                <button @click="handleSave" class="bg-blue-600 hover:bg-blue-700 text-white px-5 py-1.5 rounded text-sm font-medium shadow">儲存</button>

            </div>

        </div>

        <div class="border rounded-xl divide-y bg-white overflow-hidden max-h-[60vh] overflow-y-auto">

            <div v-if="displayNodes.length === 0" class="p-8 text-center text-gray-400 text-sm">

                📭 目前沒有資料，請新增或匯入 CSV 檔案

            </div>

           

            <div v-for="node in displayNodes" :key="node.uniqueKey"

                 @click="handleNodeClick(node)"

                 class="p-3 hover:bg-gray-50 cursor-pointer transition select-none flex justify-between items-center"

                 :class="{

                     'pl-4 font-semibold text-gray-900 bg-gray-50/50': node.level === 0,

                     'pl-10 text-gray-700': node.level === 1,

                     'pl-16 text-gray-600 bg-blue-50/20': node.level === 2

                 }">

               

                <div class="text-sm">

                    <span v-if="node.type === 0">{{ node.isExpanded ? '▼ 📁' : '▶ 📁' }} {{ node.text }}</span>

                    <span v-else-if="node.type === 1">{{ node.isExpanded ? '▼ 📂' : '▶ 📂' }} {{ node.text }}</span>

                    <span v-else>🔹 {{ node.text }} <span class="text-xs text-gray-400 ml-2" v-if="node.remark">[{{ node.remark }}]</span></span>

                </div>

                <button v-if="node.type === 2" @click.stop="handleDelete(node.dbId)"

                        class="text-red-500 hover:text-red-700 p-1 rounded hover:bg-red-50 text-xs transition">

                    🗑️ 刪除

                </button>

            </div>

        </div>

    </div>

    <script>

        const { createApp, ref, computed, watch, onMounted } = Vue;

        createApp({

            setup() {

                // 資料來源：優先從瀏覽器本地暫存取，沒有就放預設測試資料

                const rawItems = ref([]);

                const keyword = ref("");

                const isFormVisible = ref(true);

                const editingId = ref(null);

                const expandedKeys = ref(new Set());

                const form = ref({ name: "", subName: "", content: "", remark: "" });

                // 讀取快取

                onMounted(() => {

                    const localData = localStorage.getItem("sop_web_data");

                    if (localData) {

                        rawItems.value = JSON.parse(localData);

                    } else {

                        // 預設測試資料

                        rawItems.value = [

                            { id: 1, name: "電工維修", subName: "常用工具", content: "螺絲起子", remark: "工具" },

                            { id: 2, name: "電工維修", subName: "常用工具", content: "三用電錶", remark: "儀器" },

                            { id: 3, name: "軟體開發", subName: "Git指令", content: "git commit -m", remark: "基礎" }

                        ];

                    }

                });

                // 監聽資料變化，自動存入本地瀏覽器防掉檔

                watch(rawItems, (newVal) => {

                    localStorage.setItem("sop_web_data", JSON.stringify(newVal));

                }, { deep: true });

                // 核心邏輯：將扁平資料動態組合為樹狀結構 (完全對應 Android 的演算法)

                const displayNodes = computed(() => {

                    // 1. 過濾搜尋關鍵字

                    const kw = keyword.value.trim().toLowerCase();

                    const filtered = rawItems.value.filter(item =>

                        item.name.toLowerCase().includes(kw) ||

                        item.subName.toLowerCase().includes(kw) ||

                        item.content.toLowerCase().includes(kw) ||

                        item.remark.toLowerCase().includes(kw)

                    );

                    // 2. 排序

                    filtered.sort((a, b) => a.name.localeCompare(b.name) || a.subName.localeCompare(b.subName) || a.content.localeCompare(b.content));

                    // 3. 組裝 Master Nodes

                    const masterNodes = [];

                    const groupedByMain = {};

                    filtered.forEach(item => {

                        if (!groupedByMain[item.name]) groupedByMain[item.name] = {};

                        if (!groupedByMain[item.name][item.subName]) groupedByMain[item.name][item.subName] = [];

                        groupedByMain[item.name][item.subName].push(item);

                    });

                    for (const mainName in groupedByMain) {

                        const mainKey = `main:${mainName}`;

                        masterNodes.push({

                            uniqueKey: mainKey, parentKey: null, text: mainName, type: 0, level: 0,

                            isExpanded: expandedKeys.value.has(mainKey) || kw !== ""

                        });

                        for (const subName in groupedByMain[mainName]) {

                            const subKey = `sub:${mainName}:${subName}`;

                            masterNodes.push({

                                uniqueKey: subKey, parentKey: mainKey, text: subName, type: 1, level: 1,

                                isExpanded: expandedKeys.value.has(subKey) || kw !== ""

                            });

                            groupedByMain[mainName][subName].forEach(item => {

                                masterNodes.push({

                                    uniqueKey: `content:${item.id}`, parentKey: subKey, text: item.content, remark: item.remark,

                                    type: 2, level: 2, dbId: item.id, rawItem: item

                                });

                            });

                        }

                    }

                    // 4. 計算折疊展開展開後，真正要顯示出來的 displayNodes

                    const result = [];

                    function addVisibleChildren(parentKey) {

                        const children = masterNodes.filter(n => n.parentKey === parentKey);

                        children.forEach(child => {

                            result.push(child);

                            if (child.isExpanded && child.type !== 2) {

                                addVisibleChildren(child.uniqueKey);

                            }

                        });

                    }

                    addVisibleChildren(null);

                    return result;

                });

                // 點擊事件處理

                const handleNodeClick = (node) => {

                    if (node.type === 2) {

                        // 點擊最內層 -> 載入表單進行編輯

                        editingId.value = node.dbId;

                        form.value = { ...node.rawItem };

                        isFormVisible.value = true;

                    } else {

                        // 點擊目錄 -> 切換展開狀態

                        node.isExpanded = !node.isExpanded;

                        if (node.isExpanded) {

                            expandedKeys.value.add(node.uniqueKey);

                        } else {

                            expandedKeys.value.delete(node.uniqueKey);

                        }

                    }

                };

                // 儲存與更新

                const handleSave = () => {

                    if (!form.value.name || !form.value.subName || !form.value.content) {

                        alert("⚠️ 前三個欄位為必填！");

                        return;

                    }

                    if (editingId.value === null) {

                        // 新增

                        const newId = rawItems.value.length > 0 ? Math.max(...rawItems.value.map(i => i.id)) + 1 : 1;

                        rawItems.value.push({ id: newId, ...form.value });

                    } else {

                        // 更新

                        const idx = rawItems.value.findIndex(i => i.id === editingId.value);

                        if (idx !== -1) rawItems.value[idx] = { id: editingId.value, ...form.value };

                    }

                    clearForm();

                };

                // 刪除

                const handleDelete = (id) => {

                    if (confirm("確定要刪除這筆資料嗎？")) {

                        rawItems.value = rawItems.value.filter(item => item.id !== id);

                        if (editingId.value === id) clearForm();

                    }

                };

                const clearForm = () => {

                    editingId.value = null;

                    form.value = { name: "", subName: "", content: "", remark: "" };

                };

                // 📥 CSV 匯入邏輯 (純前端解析)

                const handleImportCSV = (event) => {

                    const file = event.target.files[0];

                    if (!file) return;

                    const reader = new FileReader();

                    reader.onload = (e) => {

                        const text = e.target.result;

                        const lines = text.split("\n");

                        const newItems = [];

                       

                        // 略過第一行 Header，遍歷每一行

                        for (let i = 1; i < lines.length; i++) {

                            const line = lines[i].trim();

                            if (!line) continue;

                            // 簡易 CSV 逗號切割 (若欄位內有逗號建議用 PapaParse 套件)

                            const cols = line.split(",");

                            if (cols.length >= 4) {

                                newItems.push({

                                    id: parseInt(cols[0]) || i,

                                    name: cols[1] || "",

                                    subName: cols[2] || "",

                                    content: cols[3] || "",

                                    remark: cols[4] || ""

                                });

                            }

                        }

                       

                        if (newItems.length > 0) {

                            if (confirm(`檢查到 ${newItems.length} 筆資料，是否覆蓋現有資料庫？`)) {

                                rawItems.value = newItems;

                                alert("📥 資料庫匯入成功！");

                            }

                        } else {

                            alert("❌ 找不到有效的資料結構，請確認 CSV 格式。");

                        }

                    };

                    reader.readAsText(file, "UTF-8");

                    event.target.value = ""; // 清空 input 以便重複觸發

                };

                // 📤 CSV 匯出邏輯 (帶有時間戳記的下載)

                const handleExportCSV = () => {

                    let csvContent = "\uFEFFid,name,subName,content,remark\n"; // \uFEFF 確保 Excel 開啟不亂碼

                    rawItems.value.forEach(item => {

                        csvContent += `${item.id},${item.name},${item.subName},${item.content},${item.remark}\n`;

                    });

                    const blob = new Blob([csvContent], { type: "text/csv;charset=utf-8;" });

                    const url = URL.createObjectURL(blob);

                   

                    // 產生時間戳記檔名

                    const now = new Date();

                    const timeStamp = now.getFullYear() +

                                      String(now.getMonth() + 1).padStart(2, '0') +

                                      String(now.getDate()).padStart(2, '0');

                   

                    const link = document.createElement("a");

                    link.setAttribute("href", url);

                    link.setAttribute("download", `SOP_Backup_${timeStamp}.csv`);

                    link.style.visibility = 'hidden';

                    document.body.appendChild(link);

                    link.click();

                    document.body.removeChild(link);

                };

                return {

                    rawItems, keyword, isFormVisible, editingId, form, displayNodes,

                    handleNodeClick, handleSave, handleDelete, clearForm, handleImportCSV, handleExportCSV

                };

            }

        }).mount('#app');

    </script>
